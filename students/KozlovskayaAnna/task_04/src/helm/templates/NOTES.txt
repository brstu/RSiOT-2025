1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ include "monitoring-app.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "monitoring-app.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ include "monitoring-app.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ include "monitoring-app.namespace" . }} svc -w {{ include "monitoring-app.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ include "monitoring-app.namespace" . }} {{ include "monitoring-app.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "monitoring-app.namespace" . }} -l "app.kubernetes.io/name={{ include "monitoring-app.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ include "monitoring-app.namespace" . }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ include "monitoring-app.namespace" . }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Access metrics at:
   http://{{ (index .Values.ingress.hosts 0).host }}/metrics
   or use port-forward:
   kubectl --namespace {{ include "monitoring-app.namespace" . }} port-forward svc/{{ include "monitoring-app.fullname" . }} 8080:{{ .Values.service.port }}

3. Check ServiceMonitor:
   kubectl get servicemonitor -n {{ .Values.metrics.serviceMonitor.namespace }} {{ include "monitoring-app.fullname" . }}

4. Check PrometheusRule:
   kubectl get prometheusrule -n {{ .Values.prometheusRule.namespace }} {{ include "monitoring-app.fullname" . }}

   - Group: {{ .Values.student.group }}
   - Variant: {{ .Values.student.variant }}
   - Slug: {{ .Values.student.slug }}
